2. Program using elif

a_password = "adminadmin"
password = input("Enter admin password: ")
if password == a_password:
    print("Welcome Admin!")
    role = input("Enter your role: ")
    if role == "python developer":
        print("Welcome Python Developer")
    elif role == "java developer":
        print("Welcome Java Developer")
    elif role == "web developer":
        print("Welcome Web Developer")
    elif role == "data analyst":
        print("Welcome Data Analyst")
    elif role == "sales person":
        print("Welcome Sales Person")
    else:
        print("Invalid Role")
else:
    print("Invalid Password")
